AP.AP_INVOICES_INTERFACE.INVOICE_NUM, 
AP.AP_INVOICES_INTERFACE.INVOICE_DATE,
AP_SUPPLIERS.VENDOR_NAME,
AP_SUPPLIER_SITES_ALL.VENDOR_SITE_CODE, 
AP.AP_INVOICES_INTERFACE.INVOICE_CURRENCY_CODE,
AP.AP_INVOICES_INTERFACE.INVOICE_AMOUNT, 
AP_INVOICE_DISTRIBUTIONS_ALL.INVOICE_LINE_NUMBER, 
M.SEGMENT1,
AP_INVOICE_LINES_ALL.ITEM_DESCRIPTION, 
AP_INVOICE_LINES_ALL.TAX, 
ZX_LINES.TAX_AMT, 
H.ORGANIZATION_ID,
H.NAME,
AP_INVOICES_ALL.PAYMENT_STATUS_FLAG
FROM AP_INVOICE_LINES_ALL,
           AP_INVOICES_ALL,
           AP.AP_INVOICES_INTERFACE, 
           AP.AP_INVOICE_DISTRIBUTIONS_ALL,
           AP_SUPPLIER_SITES_ALL,
           HR_OPERATING_UNITS H,
           MTL_SYSTEM_ITEMS_B M,
           AP_SUPPLIERS,
           ZX_LINES
WHERE   
         AP.AP_INVOICES_INTERFACE.INVOICE_ID=AP_INVOICE_LINES_ALL.INVOICE_ID 
AND AP_INVOICE_LINES_ALL.INVOICE_ID=AP.AP_INVOICE_DISTRIBUTIONS_ALL .INVOICE_ID
AND AP_INVOICE_LINES_ALL.INVOICE_ID=AP_INVOICES_ALL .INVOICE_ID
AND AP.AP_INVOICES_INTERFACE.VENDOR_ID=AP_SUPPLIERS.VENDOR_ID
AND AP_SUPPLIERS.VENDOR_ID=AP_SUPPLIER_SITES_ALL.VENDOR_ID
AND AP_INVOICE_LINES_ALL.LINE_NUMBER=ZX_LINES.TRX_LINE_NUMBER
AND AP_INVOICE_LINES_ALL.ORG_ID=H.ORGANIZATION_ID
AND AP_INVOICE_LINES_ALL.INVENTORY_ITEM_ID=M.INVENTORY_ITEM_ID
AND H.NAME= :P_OU_NAME
AND AP_INVOICES_ALL.PAYMENT_STATUS_FLAG='Y'
