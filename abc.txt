SELECT I1.INVOICE_ID,
I1.INVOICE_NUM,
I1.INVOICE_DATE,
I1.VENDOR_NAME,
I1.VENDOR_SITE_ID,
I2.INVOICE_ID,
I2.TAX_CODE_ID,
I2.INVOICE_LINE_NUMBER,
I3.INVOICE_ID,
I3.INVOICE_CURRENCY_CODE,
I3.INVOICE_AMOUNT,
I3.TAX_AMOUNT,
I4.INVOICE_ID,
I4.ITEM_DESCRIPTION,
I4.ORG_ID,
H.ORGANIZATION_ID,
H.NAME 
FROM HR_OPERATING_UNITS H,
	AP_INVOICES_INTERFACE I1,
	AP_INVOICE_DISTRIBUTIONS_ALL I2,
	AP_INVOICES_ALL I3,
	AP_INVOICE_LINES_ALL I4 
WHERE I1.INVOICE_ID = I2.INVOICE_ID 
AND I2.INVOICE_ID = I3.INVOICE_ID 
AND I3.INVOICE_ID = I4.INVOICE_ID 

AND I4.ORG_ID = H.ORGANIZATION_ID 
AND H.NAME = :P_OU_NAME
